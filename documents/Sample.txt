<!DOCTYPE html> <!-- tells the browser which html version to expect -->
<html lang="en"> <!-- tells the browser that this page is in english -->
	<head> <!-- used to perform preprocessing, much like a the include statements in c++ -->
		<meta charset='UTF-8'> <!-- Uniform Transformation Format - 8 bit, sets the encoding for the page -->
		<meta name="viewport" content="width=device-width, initial-scale=1.0"> <!--  this is needed to grab the dimensions of the screen, which we'll want for mobile sites -->
		<link rel="stylesheet" type="text/css" href="css/styles.css"> <!-- links a css file named styles that is located in the folder named css -->
		<title>Company XYZ</title> <!-- this part will display in on our tab -->
		<script></script> <!-- you can use this to write javascript code, or to import a javascript file -->
		
	</head>

	<!--  navigation bar -->
	<nav>
		<h1>Company XYZ</h1> <!-- h tags are used for headers and go from size 1 to 6, with 1 being the largest -->
		<hr /> <!-- create a line, basically just renders a flat border -->
		<a href="#">Home</a><!-- anchor tags are for links, the hashtag is used to set it as a dummy link -->
		<a href="#">About</a><!-- can use target="_blank" for opening in a new window / tab -->
		<a href="#">Contact</a>
		<a href="#">Apply</a>
	</nav>

	<!-- this is the main part that the people will see rendered -->
	<body>
		<br /> <!-- line break -->
		&nbsp; <!-- no backspace ASCII code, prevents html from creating a backspace at this character's position -->
		
		<!-- divider, is a block with break lines -->
		<div>This is a divider</div>
		
		<!-- paragraph -->
		<p>This is a paragraph <span style="color:blue;">with a span</span></p> <!-- span is inline so it causes no format change, it's perfect for applying styles in mid paragraph -->
		
		<!-- abbreviation is used to create tooltips that display the title upon hover -->
		<p><abbr title="Association for Computing Machinery">ACM</abbr></p>
		
		<!-- image -->
		<img src="#" alt="Description Here" height="42" width="42" />
		
		<!-- iframes are used to embed things, such as webpages, videos, and documents -->
		<iframe src="#"></iframe>
		
		
		<!-- a table, much like excel spreadsheets -->
		<table>
			<!-- table row -->
			<tr>
				<!-- table headers -->
				<th>Col 1</th>
				<th>Col 2</th>
			</tr>
			<tr>
				<!-- table data -->
				<td>item X</td>
				<td>item Y</td>
			</tr>
		</table>
		
		<!-- unordered list, uses dots -->
		<ul>
			<!-- list items -->
			<li>item 1</li>
			<li>item 2</li>
			<li>item 3</li>
		</ul>
		
		<!-- ordered list, uses types 1 for numbering, A for lettering, and I for roman numerals -->
		<ol type="I">
			<li>item 1</li>
			<li>item 2</li>
			<li>item 3</li>
		</ol>
	</body>

	<!-- the footer is a place that will typically contain contact info, give credit, publishing info, provide links to less advertised portions of the site, or external sites -->
	<footer></footer>
</html>
